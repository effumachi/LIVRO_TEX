\noindent 1. Dada a função $f(x)=4x+6$, determine:
\begin{itemize}
	\item[a.]{$f(1)$}
	\item[b.]{$f(2)$}
	\item[c.]{$f(-1)$}
	\item[d.]{$f(-4)$}
	\item[e.]{$f(0)$}
	\item[f.]{$f$ é crescente ou decrescente? Por quê?}
\end{itemize}

	\item[d.]{$f(-4)$}
	\item[e.]{$f(0)$}

	\item[c.]{$f(x)=6x-15$}
	\item[d.]{$f(x)=-10x-30$}
\end{itemize}
